### Example

create github action in `.github/workflows/outdated.yml`

```yml
name: Outdated Version Action

on:
  pull_request:
    types: [opened]

jobs:
  outdated-version:
    runs-on: ubuntu-latest
    steps:
      - name: outdated version
        uses: ycjcl868/outdated-version-action@v1
        with:
          token: ${{ secrets.GITHUB_TOKEN }}
```
